import React from 'react';

/** Button usage

```jsx
<Button>Reserve a table</Button>
<Button variant="secondary" size="lg">Plan a stay</Button>
<Button variant="ghost" size="sm">View menu</Button>
```

- `variant`: `primary` (terracotta), `secondary` (indigo), `ghost` (hairline).
- `size`: `sm` / `md` / `lg`.
- Copy is always sentence-case in the label but renders uppercase + tracked.
- Hover lifts one shadow level and darkens ~6%; press has no color flash.
*/
